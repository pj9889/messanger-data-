function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var logSheet = ss.getSheetByName("Logs") || ss.insertSheet("Logs"); var dailySheet = ss.getSheetByName("DailyStats") || ss.insertSheet("DailyStats"); // DailyStats শিট সেটআপ if (dailySheet.getRange("A1").getValue() === "") { dailySheet.appendRow(["Date", "Visit", "Click"]); } var data = JSON.parse(e.postData.contents); var eventType = data.event; var ip = data.ip || "Unknown"; var today = new Date().toLocaleDateString(); // আজকের তারিখ (e.g. 5/9/2026) // ১. Logs শিটে এন্ট্রি (কান্ট্রি ডিটেকশনসহ) var country = "Unknown"; try { var res = UrlFetchApp.fetch("http://ip-api.com/json/" + ip); var geo = JSON.parse(res.getContentText()); if(geo.status === "success") country = geo.country; } catch(err) {} logSheet.appendRow([new Date(), eventType, country, ip]); // ২. প্রতিদিনের ডাটা আপডেট লজিক var dailyData = dailySheet.getDataRange().getValues(); var rowIndex = -1; for (var i = 1; i < dailyData.length; i++) { if (dailyData[i][0] == today) { rowIndex = i + 1; break; } } if (rowIndex === -1) { // নতুন দিনের জন্য নতুন রো var visit = (eventType === "Total_Visit") ? 1 : 0; var click = (eventType === "Button_Click") ? 1 : 0; dailySheet.appendRow([today, visit, click]); } else { // আগের রো আপডেট var col = (eventType === "Total_Visit") ? 2 : 3; var currentVal = dailySheet.getRange(rowIndex, col).getValue(); dailySheet.getRange(rowIndex, col).setValue(Number(currentVal) + 1); } return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var dailySheet = ss.getSheetByName("DailyStats"); var logsSheet = ss.getSheetByName("Logs"); if (e.parameter.action === 'getStats') { var dateParam = e.parameter.date || new Date().toLocaleDateString(); var dailyData = dailySheet.getDataRange().getValues(); var visit = 0, click = 0; for (var i = 1; i < dailyData.length; i++) { if (dailyData[i][0] == dateParam) { visit = dailyData[i][1]; click = dailyData[i][2]; break; } } // কান্ট্রি ডাটা (Logs থেকে) var logs = logsSheet.getDataRange().getValues(); var countryMap = {}; var total = 0; for (var j = 1; j < logs.length; j++) { var logDate = new Date(logs[j][0]).toLocaleDateString(); if (logDate == dateParam) { var c = logs[j][2]; if(c && c !== "Unknown") { countryMap[c] = (countryMap[c] || 0) + 1; total++; } } } var sortedCountries = Object.keys(countryMap).sort((a,b) => countryMap[b] - countryMap[a]); return ContentService.createTextOutput(JSON.stringify({ visits: visit, clicks: click, topCountry: sortedCountries[0] || "N/A", countries: sortedCountries.slice(0, 5), countryCounts: sortedCountries.slice(0, 5).map(name => ((countryMap[name] / total) * 100).toFixed(1)) })).setMimeType(ContentService.MimeType.JSON); } }
